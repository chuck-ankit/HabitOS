const GOOGLE_CLIENT_ID = process.env.VITE_GOOGLE_CLIENT_ID || 'YOUR_GOOGLE_CLIENT_ID'; const DRIVE_FILE_NAME = 'habitos-backup-test.json'; let tokenClient = null; let accessToken = null; let gapiInitialized = false; let gisInitialized = false; console.log('\n========================================'); console.log('🔍 HabitOS Google API Test Script'); console.log('========================================\n'); if (!GOOGLE_CLIENT_ID || GOOGLE_CLIENT_ID === 'YOUR_GOOGLE_CLIENT_ID' || GOOGLE_CLIENT_ID === '') { console.log('❌ ERROR: VITE_GOOGLE_CLIENT_ID is not configured!'); console.log('\n📝 To fix:'); console.log(' 1. Copy .env.example to .env'); console.log(' 2. Add your Google Client ID to .env'); console.log(' 3. Run: VITE_GOOGLE_CLIENT_ID=your_id node test-google-auth.cjs\n'); console.log(' Or add it to your .env file and restart the dev server.\n'); process.exit(1); } console.log(`✅ Client ID found: ${GOOGLE_CLIENT_ID.substring(0, 30)}...\n`); // Load Google APIs dynamically const loadScript = (src) => new Promise((resolve, reject) => { if (document.querySelector(`script[src="${src}"]`)) { resolve(); return; } const script = document.createElement('script'); script.src = src; script.onload = resolve; script.onerror = reject; document.body.appendChild(script); }); async function initGoogleApi() { return new Promise((resolve, reject) => { if (gapiInitialized && gisInitialized) { resolve(); return; } loadScript('https://apis.google.com/js/api.js').then(() => { window.gapi.load('client', async () => { await window.gapi.client.init({ discoveryDocs: ['https://www.googleapis.com/discovery/v1/apis/drive/v3/rest'], }); gapiInitialized = true; checkReady(); }); }).catch(reject); loadScript('https://accounts.google.com/gsi/client').then(() => { tokenClient = window.google.accounts.oauth2.initTokenClient({ client_id: GOOGLE_CLIENT_ID, scope: 'https://www.googleapis.com/auth/drive.file', callback: (response) => { if (response.access_token) { accessToken = response.access_token; } }, }); gisInitialized = true; checkReady(); }).catch(reject); function checkReady() { if (gapiInitialized && gisInitialized) { resolve(); } } }); } function signInWithGoogle() { return new Promise((resolve, reject) => { if (!tokenClient) { reject(new Error('Google API not initialized')); return; } tokenClient.callback = async (response) => { if (response.access_token) { accessToken = response.access_token; resolve(response); } else { reject(new Error('Failed to get access token')); } }; try { tokenClient.requestAccessToken({ prompt: 'consent' }); } catch (error) { reject(error); } }); } function isSignedIn() { return !!accessToken; } async function getUserInfo() { if (!accessToken) return null; try { const response = await fetch('https://www.googleapis.com/oauth2/v2/userinfo', { headers: { Authorization: `Bearer ${accessToken}` }, }); return await response.json(); } catch (error) { console.error('Error getting user info:', error); return null; } } async function saveToGoogleDrive(data) { if (!accessToken) throw new Error('Not signed in to Google'); const jsonContent = JSON.stringify(data, null, 2); const metadata = { name: DRIVE_FILE_NAME, mimeType: 'application/json' }; // Try to find existing file const searchResponse = await window.gapi.client.drive.files.list({ q: `name = '${DRIVE_FILE_NAME}' and trashed = false`, fields: 'files(id, name)', }); if (searchResponse.result.files?.length > 0) { // Update existing file const fileId = searchResponse.result.files[0].id; const form = new FormData(); form.append('metadata', new Blob([JSON.stringify(metadata)], { type: 'application/json' })); form.append('file', new Blob([jsonContent], { type: 'application/json' })); await fetch(`https://www.googleapis.com/upload/drive/v3/files/${fileId}?uploadType=multipart`, { method: 'PATCH', headers: { Authorization: `Bearer ${accessToken}` }, body: form, }); return fileId; } else { // Create new file const form = new FormData(); form.append('metadata', new Blob([JSON.stringify(metadata)], { type: 'application/json' })); form.append('file', new Blob([jsonContent], { type: 'application/json' })); const response = await fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart', { method: 'POST', headers: { Authorization: `Bearer ${accessToken}` }, body: form, }); const result = await response.json(); return result.id; } } async function loadFromGoogleDrive() { if (!accessToken) throw new Error('Not signed in to Google'); const searchResponse = await window.gapi.client.drive.files.list({ q: `name = '${DRIVE_FILE_NAME}' and trashed = false`, fields: 'files(id, name)', }); if (!searchResponse.result.files?.length) { return null; } const fileId = searchResponse.result.files[0].id; const response = await fetch(`https://www.googleapis.com/drive/v3/files/${fileId}?alt=media`, { headers: { Authorization: `Bearer ${accessToken}` }, }); return await response.json(); } // Run tests in browser async function runTests() { let passed = 0; let failed = 0; // Test 1: Initialize Google API console.log('📋 Test 1: Initialize Google API...'); try { await initGoogleApi(); console.log('✅ PASSED: Google API initialized\n'); passed++; } catch (err) { console.log(`❌ FAILED: ${err.message}\n`); failed++; alert('Failed to initialize Google API. Check console for details.'); return; } // Test 2: Check sign-in status console.log('📋 Test 2: Check sign-in status...'); const signedIn = isSignedIn(); if (signedIn) { console.log('✅ Already signed in!\n'); } else { console.log('⚠️ Not signed in\n'); } passed++; // Test 3: Sign in if (!signedIn) { console.log('📋 Test 3: Sign in with Google...'); console.log(' A browser popup will open. Please sign in...\n'); try { await signInWithGoogle(); console.log('✅ PASSED: Signed in successfully!\n'); passed++; } catch (err) { console.log(`❌ FAILED: ${err.message}\n`); failed++; alert('Sign-in was cancelled or failed.'); return; } } // Test 4: Get user info console.log('📋 Test 4: Get user info...'); const user = await getUserInfo(); if (user && user.email) { console.log(` User: ${user.name}`); console.log(` Email: ${user.email}`); console.log('✅ PASSED: Got user info\n'); passed++; } else { console.log('❌ FAILED: Could not get user info\n'); failed++; } // Test 5: Save to Drive console.log('📋 Test 5: Save test data to Google Drive...'); try { const testData = { test: true, timestamp: new Date().toISOString(), message: 'HabitOS API Test' }; const fileId = await saveToGoogleDrive(testData); console.log(` File saved with ID: ${fileId}`); console.log('✅ PASSED: Data saved to Google Drive\n'); passed++; } catch (err) { console.log(`❌ FAILED: ${err.message}\n`); failed++; } // Test 6: Load from Drive console.log('📋 Test 6: Load test data from Google Drive...'); try { const loaded = await loadFromGoogleDrive(); if (loaded && loaded.test === true) { console.log(` Loaded timestamp: ${loaded.timestamp}`); console.log('✅ PASSED: Data loaded from Google Drive\n'); passed++; } else { console.log('❌ FAILED: Invalid data loaded\n'); failed++; } } catch (err) { console.log(`❌ FAILED: ${err.message}\n`); failed++; } // Summary console.log('========================================'); console.log('📊 Test Summary'); console.log('========================================'); console.log(` ✅ Passed: ${passed}`); console.log(` ❌ Failed: ${failed}`); console.log('========================================\n'); if (failed === 0) { alert('🎉 All tests passed! Google integration is working correctly.'); console.log('🎉 All tests passed! Google integration is working correctly.\n'); } else { alert(`⚠️ ${failed} test(s) failed. Check console for details.`); console.log('⚠️ Some tests failed. Please check the errors above.\n'); } } // Auto-run when script is loaded in browser console.log('✅ Test script loaded. Starting tests...\n'); runTests();
